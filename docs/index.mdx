import { Tag, TagIntent } from "../app-src/components/Tag";

## Examples

```tsx
<Tag intent={TagIntent.Default}>A component</Tag>

// or for default:
<Tag>A component</Tag>
```

| Intent              | Example                                       |
| ------------------- | --------------------------------------------- |
| `TagIntent.Default` | <Tag intent={TagIntent.Default}>default</Tag> |
| `TagIntent.Warning` | <Tag intent={TagIntent.Warning}>warning</Tag> |
| `TagIntent.Error`   | <Tag intent={TagIntent.Error}>error</Tag>     |

### Special

```tsx
<Tag intent={TagIntent.Comment}>
    This is a comment with a <Tag>tag</Tag> inside
</Tag>
```
